SELECT ISSUES.SUMMARY, IL.NAME FROM ISSUES JOIN ISSUESLISTS IL on ISSUES.ISSUESLISTS_ID = IL.ID;
SELECT ISSUES.SUMMARY, USERS.FIRSTNAME, USERS.LASTNAME FROM ISSUES JOIN USERS ON ISSUES.USER_ID_ASSIGNEDTO_ID = USERS.ID;
SELECT USERS.FIRSTNAME, USERS.LASTNAME, COUNT(*) AS TASKS_ASSIGNED FROM ISSUES JOIN USERS ON ISSUES.USER_ID_ASSIGNEDTO_ID = USERS.ID GROUP BY USERS.FIRSTNAME, USERS.LASTNAME;
